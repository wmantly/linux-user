Quick Start

const linuxUser = require('linux-sys-user');

// Add a new user
linuxUser.addUser({
    username: "john",
    create_home: true,
    shell: "/bin/bash"
}, function(err, user) {
    if (err) {
        console.error('Error:', err);
    } else {
        console.log('User created:', user);
    }
});

// Using promises
const linuxUserPromise = require('linux-sys-user').promise();

async function createUser() {
    try {
        const user = await linuxUserPromise.addUser({
            username: "john",
            create_home: true,
            shell: "/bin/bash"
        });
        console.log('User created:', user);
    } catch (err) {
        console.error('Error:', err);
    }
}
